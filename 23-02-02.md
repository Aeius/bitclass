### function
```
// 호출 먼저 가능 단, 스크립트 단위로 컴파일되기때문에
// 호출과 선언이 다른 스크립트일 경우 선언이 먼저 되야함
var val = func1();

function func1() {
    console.log("func1 run...");
    return 1234; // 생략가능
}

f(); // func1 run...

// f2(); 이건 안됨

    console.log("run...");
};
f2(); // run...

(function(){
    console.log("rruunn...");
```

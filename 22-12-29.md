#### Listener Adapter
실제 사용은 각 기능을 구현할 때 Listener의 모든 기능이 필요하지 않기 때문에 interface를 구현하게 될 경우 필요없는 메서드도 정의 해줘야 하기 때문에    
코드가 지저분해진다. Adapter를 통해 익명클래스를 구현하는 방식으로 하게되면 필요한 기능만 재정의 해서 불필요한 코드량을 줄일 수 있다.
```
public class Ex16 extends JFrame{
	public Ex16() {
		addWindowListener(new WindowAdapter() { // 닫기버튼 구현
			@Override
			public void windowClosing(WindowEvent e) {
				dispose();
			}
		});
		setDefaultCloseOperation(JFrame.DISPOSE_ON_CLOSE); // JFrame은 닫기버튼을 미리 구현해뒀다.
		setBounds(500, 100, 300, 300);
		setVisible(true);
	}

	public static void main(String[] args) {
		Ex16 me = new Ex16();
	}

}

```
package com.bit;

import java.awt.BorderLayout;

import java.io.BufferedOutputStream;
import java.io.File;
import java.io.FileInputStream;
import java.io.FileNotFoundException;
import java.io.FileOutputStream;
import java.io.IOException;
import java.io.InputStream;
import java.io.OutputStream;

public class Notepad extends Frame {
	int width = 700;
	int height = 500;

	public Notepad() {
		setTitle("제목없음");
		// 메모장 입력 할 TextArea 붙이기
		TextArea ta = new TextArea(width, height);
		add(ta);
		// 메뉴바 - 파일
		MenuBar mb = new MenuBar();
		Menu m1 = new Menu("파일");
		mb.add(m1);
		// 새로만들기 버튼 구현
		MenuItem mi1 = new MenuItem("새로만들기");
		mi1.addActionListener(new ActionListener() {
			@Override
			public void actionPerformed(ActionEvent e) {
				// 적은 텍스트가 없을 경우
				if (ta.getText().isEmpty()) {
					setTitle("제목없음");
					// 하나라도 적은 텍스트가 있는 경우 (저장요청)
				} else {
					FileDialog dia2 = new FileDialog(new Frame(), "저장", FileDialog.SAVE);
					dia2.setTitle("저장");
					dia2.setLocation(100, 100);
					dia2.setVisible(true);
					// FileDialog에서 입력한 제목을 토대로 파일 생성
					File file = new File(dia2.getFile());
					try {
						// 새로운 파일을 저장 할 경우
						if (!file.exists()) {
							File newFile = new File(dia2.getFile() + ".txt");
							newFile.createNewFile();
							OutputStream os = null;
							BufferedOutputStream bos = null;
							os = new FileOutputStream(newFile);
							bos = new BufferedOutputStream(os);

							// 작성된 문자열 저장
							String msg = ta.getText();
							for (int i = 0; i < msg.length(); i++) {
								bos.write(msg.charAt(i));
							}

							if (bos != null)
								bos.close();
							if (os != null)
								os.close();
							// 기존의 데이터들 파일에 저장 후
							// 현재 UI는 초기화 되어야하므로 초기화 하는 작업
							ta.setText("");
							setTitle("제목없음");

							// 기존 파일에 저장 혹은 덮어쓰기 하는 경우
						} else {
							OutputStream os = null;
							BufferedOutputStream bos = null;
							os = new FileOutputStream(file);
							bos = new BufferedOutputStream(os);

							// 작성된 문자열 저장
							String msg = ta.getText();
							for (int i = 0; i < msg.length(); i++) {
								bos.write(msg.charAt(i));
							}
							if (bos != null)
								bos.close();
							if (os != null)
								os.close();
							// 기존의 데이터들 파일에 저장 후
							// 현재 UI는 초기화 되어야하므로 초기화 하는 작업
							ta.setText("");
							setTitle("제목없음");
						}
					} catch (FileNotFoundException e1) {
						e1.printStackTrace();
					} catch (IOException e1) {
						e1.printStackTrace();
					}
				}
			}
		});
		m1.add(mi1);
		// 열기 버튼 구현
		MenuItem mi2 = new MenuItem("열기");
		mi2.addActionListener(new ActionListener() {
			@Override
			public void actionPerformed(ActionEvent e) {
				FileDialog dia2 = new FileDialog(new Frame(), "열기", FileDialog.LOAD);
				dia2.setTitle("열기");
				dia2.setLocation(100, 100);
				dia2.setVisible(true);
				// FileDialog에서 입력한 제목을 토대로 파일 생성
				File file = new File(dia2.getFile());
				try {
					InputStream is = null;
					BufferedInputStream bis = null;
					is = new FileInputStream(file);
					bis = new BufferedInputStream(is);

					// 저장된 문자열 불러오기
					String msg = "";
					while (true) {
						int su = bis.read();
						if (su == -1)
							break;
						msg += ((char) su) + "";
					}
					ta.setText(msg);

					if (bis != null)
						bis.close();
					if (is != null)
						is.close();
					// 저장할 때 사용한 제목으로 타이틀 변경
					setTitle(dia2.getFile());
				} catch (FileNotFoundException e1) {
					e1.printStackTrace();
				} catch (IOException e1) {
					e1.printStackTrace();
				}

			}
		});
		m1.add(mi2);
		// 저장 버튼 구현
		MenuItem mi3 = new MenuItem("저장");
		mi3.addActionListener(new ActionListener() {
			@Override
			public void actionPerformed(ActionEvent e) {
				FileDialog dia2 = new FileDialog(new Frame(), "저장", FileDialog.SAVE);
				dia2.setLocation(100, 100);
				dia2.setVisible(true);
				// FileDialog에서 입력한 제목을 토대로 파일 생성
				File file = new File(dia2.getFile());
				try {
					// 새로운 파일을 저장 할 경우
					if (!file.exists()) {
						File newFile = new File(dia2.getFile() + ".txt");
						newFile.createNewFile();
						OutputStream os = null;
						BufferedOutputStream bos = null;
						os = new FileOutputStream(newFile);
						bos = new BufferedOutputStream(os);

						// 작성된 문자열 저장
						String msg = ta.getText();
						for (int i = 0; i < msg.length(); i++) {
							bos.write(msg.charAt(i));
						}

						if (bos != null)
							bos.close();
						if (os != null)
							os.close();

	public void open() {
		FileDialog dia2 = new FileDialog(new Frame(), "열기", FileDialog.LOAD);

			InputStream is = null;
			BufferedInputStream bis = null;

			bis = new BufferedInputStream(is);
